                              ? "border-emerald-300 bg-emerald-50/20"
                               : "border-slate-200 hover:border-slate-300 bg-slate-50/50"
                        }`}
                      >
                        <input
                          type="file"
                          id="receiptInput"
                          className="hidden"
                          accept="image/*"
                          onChange={handleFileInput}
                        />
                        <label htmlFor="receiptInput" className="cursor-pointer w-full h-full flex flex-col items-center justify-center gap-2">
                          <UploadCloud size={32} className={contingent.buktiTransferUrl ? "text-emerald-500" : "text-slate-400"} />
                          <div>
                            <span className="text-xs text-slate-700 font-extrabold block">
                              {contingent.buktiTransferUrl ? "Ganti Bukti Transfer" : "Klik atau seret bukti transfer"}
                            </span>
                            <span className="text-[10px] text-slate-400 font-semibold block mt-1">Format: JPG, JPEG, atau PNG (Maks 5MB)</span>
                          </div>
                        </label>
                      </div>
                      {contingent.buktiTransferUrl && (
                        <div className="flex items-center justify-between bg-slate-50 border border-slate-100 p-3 rounded-xl">
                          <span className="text-[10px] font-bold text-emerald-600 flex items-center gap-1">
                            <CheckCircle2 size={12} /> Bukti terupload & menunggu review
                          </span>
                          <a
                            href={contingent.buktiTransferUrl}
                            target="_blank"
                            rel="noreferrer"
                            className="text-xs text-emerald-600 hover:underline font-extrabold flex items-center gap-1"
                          >
                            <ImageIcon size={12} /> Lihat
                          </a>
                        </div>
                      )}
                    </div>
                  )}
                </div>
              </div>
              <div className="pt-4 border-t border-slate-200 flex justify-end">
                <button
                  onClick={() => setIsPaymentModalOpen(false)}
                  className="bg-slate-900 hover:bg-slate-800 text-white font-extrabold text-xs px-6 py-3 rounded-xl transition-all uppercase tracking-wider"
                >
                  Tutup Rincian
                </button>
              </div>
            </div>
          </div>
        </div>
      )}
